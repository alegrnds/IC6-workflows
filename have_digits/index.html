def have_digits(s: str) -> int:
    """Checks if a string has digits in it

    Args: 
        s: string

    Returns:
        out: 0 if it doesn't contain digits and 1 if it does
    """

    out = 0

    # loop through the string
    for c in s:
        # check if the character is a digit
        if c.isdigit():
            out = 1
            break

    return out
